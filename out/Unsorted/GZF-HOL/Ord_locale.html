(*It would be nice to use the ‚Äπclass.wellorder‚Ä∫ locale, but unfortunately
  this assumes that the entire type is well-ordered by leq and lt, 
  but we just want this to be true for the ordinal portion.
  We could assume an 'Ordinal' subtype and assume an instantiation of the wellorder class on there,

locale Ord = GZF Mem Emp Set Union Subset Pow Succ Inf Repl
    for Mem :: "'d ‚áí 'd ‚áí bool" (infixl ‚Äπ‚àà‚Ä∫ 50)
    and Emp :: "'d" (‚Äπ‚àÖ‚Ä∫) and Set :: "'d ‚áí bool" 
    and Union :: "'d ‚áí 'd" (‚Äπ‚ãÉ_‚Ä∫ [90] 90)
    and Subset:: "['d,'d] ‚áí bool" (infixl ‚Äπ‚äÜ‚Ä∫ 50)
    and Pow ::  "'d ‚áí 'd" (‚Äπùí´‚Ä∫ 90)
    and Succ :: "'d ‚áí 'd"
    and Inf :: "'d"  ‚Äï ‚Äπwitness of Infinity‚Ä∫
    and Repl :: "[['d,'d] ‚áí bool] ‚áí 'd ‚áí'd" (‚Äπ‚Ñõ‚Ä∫ 90)  +
  fixes
    Ord :: "'d ‚áí bool" and
    lt :: "'d ‚áí 'd ‚áí bool" (infixl ‚Äπ<‚Ä∫ 50) and
    zero :: "'d" (‚Äπ0‚Ä∫) and
    succ :: "'d ‚áí 'd" and
    Limit :: "'d ‚áí bool" and
    predSet :: "'d ‚áí 'd" and
    OrdRec :: "('d ‚áí 'd ‚áí 'd) ‚áí ('d ‚áí 'd ‚áí 'd) ‚áí 'd ‚áí 'd ‚áí 'd" and
    omega :: "'d" (‚Äπœâ‚Ä∫)
  assumes 
    (*Axioms for the Ord predicate*)
    zero_ord [simp]: "Ord 0" and
    œâ_ord [simp]: "Ord œâ" and
    lt_ord : "i < j ‚ü∂ Ord i ‚àß Ord j" and
    ord_succ_iff [iff] : "Ord (succ i) ‚ü∑ Ord i" and
    (*Definition of 0, succ, leq and Limit*)
    zero_def : "¬¨ (i < 0)" and
    succ_def : "(‚àÄŒ≤. Ord Œ≤ ‚ü∂ (‚àÄŒ±. Œ± < (succ Œ≤) ‚ü∑ Œ± = Œ≤ ‚à® Œ± < Œ≤))" and
    limit_def : "Limit Œº ‚ü∑ (Ord Œº ‚àß 0 < Œº ‚àß (‚àÄŒ≤. Œ≤ < Œº ‚ü∂ succ Œ≤ < Œº)) " and
    (*Characterisation of < as a wellorder*)
    lt_trans : "‚àÄi j k. i < j ‚ü∂ j < k ‚ü∂ i < k" and
    lt_notsym : "‚àÄi j. i < j ‚ü∂ ¬¨ j < i" and
    lt_linear : "‚àÄi j. Ord i ‚àß Ord j ‚ü∂ i < j | i = j | j < i" and
    lt_induct : "Ord a ‚ü∂ (‚àÄx. Ord x ‚ü∂ (‚àÄy. y < x ‚ü∂ P y) ‚ü∂ P x) ‚ü∂ P a" and

    predset_def : "‚àÄŒ≤. Ord Œ≤ ‚ü∂ Set (predSet Œ≤) ‚àß (‚àÄŒ±. Œ± ‚àà predSet Œ≤ ‚ü∑ Œ± < Œ≤)" and
    ordrec_0 : "‚àÄG F A. OrdRec G F A 0 = A" and 
    ordrec_succ : "‚àÄG F A b. Ord b ‚ü∂ 
      OrdRec G F A (succ b) = F (succ b) (OrdRec G F A b)" and
    ordrec_lim :  "‚àÄG F A z. Limit z ‚ü∂ 
      OrdRec G F A z = G z (Lambda (predSet z) (ŒªŒ≤. OrdRec G F A Œ≤))"

lemma le_iff: "i ‚â§ j <-> i<j | (i=j & Ord(j))"
proof 
  assume "i ‚â§ j" hence "i < succ j" by simp
  moreover hence "Ord j" using lt_ord succ_ordD  by auto
  ultimately show "i < j ‚à® (i = j ‚àß Ord j)" using succ_def by auto
next
  assume "i < j ‚à® i = j ‚àß Ord j"
  thus "i ‚â§ j" using succ_def lt_ord by auto
qed

lemma leI: "i<j ==> i ‚â§ j"
  by (simp add: le_iff)

lemma Ord_linear_lt:
  assumes "Ord i" "Ord j"

done

  done

lemma succ_leI: "i<j ==> succ(i) ‚â§ j"
proof (rule not_lt_iff_le [THEN iffD1])
  assume "i < j"
  thus "Ord j" using lt_ord2 by auto
  from ‚Äπi < j‚Ä∫ have "Ord i" using lt_ord by simp
  thus "Ord (succ i)" using succ_ord by auto
  from ‚Äπi < j‚Ä∫ show "¬¨ j ‚â§ i" by (blast elim: leE lt_asym)
qed

  from ‚Äπsucc i ‚â§ j‚Ä∫ show "Ord i" using lt_ord succ_ordD by auto
  from ‚Äπsucc i ‚â§ j‚Ä∫ show "¬¨ j ‚â§ i" by (blast elim: leE lt_asym)
qed

lemma non_succ_LimitI:
  assumes i: "0<i" and nsucc: "‚ãÄy. succ(y) ‚â† i"
  shows "Limit(i)"
proof -
  have Oi: "Ord(i)" using i by (simp add: lt_ord2)
  { fix y
    assume yi: "y<i"
    hence Osy: "Ord(succ(y))" by (simp add: lt_ord succ_ord)
    have "~ i ‚â§ y" using yi by (blast dest: le_imp_not_lt)
    hence "succ(y) < i" using nsucc [of y]
      by (blast intro: Ord_linear_lt [OF Osy Oi]) }
  thus ?thesis using i Oi by (auto simp add: limit_def)
qed
