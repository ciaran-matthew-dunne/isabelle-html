definition the_default :: "'d ⇒ ('d ⇒ bool) ⇒ 'd" (‹ι'' _ _› [90,90] ) where
 "ι' d F ≡ if Ex1 F then (THE x. F x) else d"

lemma the_def_if : "if (Ex1 F) then (F a ⟷ ι' d F = a) 
                               else ι' d F = d"
proof (cases ‹Ex1 F›)
  assume "Ex1 F" hence eq:"ι' d F = (THE x. F x)" 
    unfolding the_default_def by auto
  have "F a ⟷ ι' d F = a" 
  proof
    assume "F a"
    thus "ι' d F = a" by (subst eq, insert ‹Ex1 F›, auto)
  next
    assume "ι' d F = a"
    thus "F a" using eq ‹Ex1 F› by (auto intro: theI)
  qed
  thus "if (Ex1 F) then (F a ⟷ ι' d F = a) else ι' d F = d" using ‹Ex1 F› by simp
next
  assume "¬ Ex1 F"
  thus "if (Ex1 F) then (F a ⟷ ι' d F = a) else ι' d F = d" 
    unfolding the_default_def by auto
qed

lemma the_def_eq : assumes "P a" "⋀x. P x ⟹ x = a"
  shows "ι' d P = a"
proof -
  from assms have "Ex1 P" by auto
  hence "P a ⟷ (ι' d P) = a" using the_def_if by auto
  thus "(ι' d P) = a" using ‹P a› by auto
qed

lemma the_defI : assumes "P a" "⋀x. P x ⟹ x = a" 
  shows "P (ι' d P)"
proof - 
  from the_def_eq assms have "ι' d P = a" by blast 
  thus "P (ι' d P)" using ‹P a› by simp
qed

lemma the_defI' : assumes "Ex1 P" shows "P (ι' d P)"
  using assms by (blast intro: the_defI)

lemma the_defI2 : assumes "P a" "⋀x. P x ⟹ x = a" "⋀x. P x ⟹ Q x"
  shows "Q (ι' d P)" by (blast intro: assms the_defI)

lemma the_defI2' : assumes "Ex1 P" "⋀x. P x ⟹ Q x" shows "Q (ι' d P)" 
  by (blast intro: assms(2) the_defI2[where P=P and Q=Q] ex1E[OF assms(1)])

lemma the_def_eq' : "⟦ Ex1 P ; P a ⟧ ⟹ (ι' d P) = a"
  by (blast intro: the_def_eq)

subsection ‹Restricted quantification using a predicate›

definition rall :: "('d ⇒ bool) ⇒ ('d ⇒ bool) ⇒ bool" where
  rall_def : "rall P Q ≡ ∀x. P x ⟶ Q x"
definition rex :: "('d ⇒ bool) ⇒ ('d ⇒ bool) ⇒ bool" where
  rex_def : "rex P Q ≡ ∃x. P x ∧ Q x"

(*Taken from ZF/OrdQuant, unfolded syntax*)

by (simp add: rall_def)

by (simp add: rall_def)

lemma rev_rallE [elim]:
    "[| rall M P;  ~ M(x) ==> Q;  P(x) ==> Q |] ==> Q"
by (simp add: rall_def, blast)

lemma rallE: "[| rall M P; P(x) ==> Q;  ~ M(x) ==> Q |] ==> Q"
by blast

lemma rall_triv [simp]: "(rall M (λx. P)) ⟷ ((∃x. M x) ⟶ P)"
by (simp add: rall_def)

lemma rall_cong [cong]:
    "(!!x. M(x) ==> P(x) <-> P'(x)) ==> (rall M P) <-> (rall M P')"

lemma rexI [intro]: "[| P(x); M(x) |] ==> rex M P"

lemma rexE [elim]: "[| rex M P;  !!x. [| M(x); P(x) |] ==> Q |] ==> Q"

    "(!!x. M(x) ==> P(x) <-> P'(x)) ==> (rex M P) <-> (rex M P')"

thm repl
lemma repl_set : "Set x ⟹ Set (Repl P x)" using repl by auto

lemma repl_iff : assumes "Set x" "ball x (λa. ∃⇩≤⇩1 b. P a b)"
  shows "b ∈ Repl P x ⟷ (bex x (λa. P a b))"
  using repl assms unfolding ball_def rall_def bex_def rex_def by auto

definition Replace :: "[['d,'d] ⇒ bool, 'd] ⇒ 'd" where
  "Replace P x ≡ Repl (λx y. (∃!z. P x z) ∧ P x y) x"

lemma replace_set : "Set x ⟹ Set (Replace P x)" 
  by (unfold Replace_def, rule repl_set)

lemma replace_iff : assumes "Set x" 
  shows "b ∈ Replace P x ⟷ (bex x (λa. P a b ∧ (∀c. P a c ⟶ c = b)))" 
proof -
  let ?Q = "(λx y. (∃!z. P x z) ∧ P x y)"
  have "ball x (λa. ∃⇩≤⇩1 b. ?Q a b)" unfolding Uniq_def by auto
  hence "b ∈ Repl ?Q x ⟷ (bex x (λa. ?Q a b))" by (rule repl_iff[OF assms])
  thus ?thesis unfolding Replace_def Ex1_def by auto
qed

lemma replaceI : "⟦ P a b ; a ∈ x ; ⋀c. P a c ⟹ c = b ⟧ ⟹ b ∈ Replace P x"
  using replace_iff setI by blast

lemma replaceE: assumes "Set x"
  shows "⟦ b ∈ Replace P x ; 
          ⋀a. ⟦a ∈ x ; P a b ; ∀c. P a c ⟶ c = b ⟧ ⟹ R ⟧
          ⟹ R"
  by (rule replace_iff [OF assms, THEN iffD1, THEN bexE], simp+)

lemma replaceE2 : assumes "Set x"
  shows "⟦ b ∈ Replace P x ; ⋀a. ⟦a ∈ x ; P a b ⟧ ⟹ R ⟧ ⟹ R"
  by (erule replaceE[OF assms], blast)

lemma replace_cong : assumes "Set x" "Set y" shows [cong]:
  "⟦x = y ; ⋀a b. a ∈ y ⟹ P a b ⟷ Q a b⟧ ⟹ Replace P x = Replace Q y"
  by (rule equality_iffI[OF replace_set[OF assms(1)] 
                            replace_set[OF assms(2)]], 
      simp add: assms replace_iff)

definition RepFun :: "['d ⇒ 'd, 'd] ⇒ 'd" where
  "RepFun F x ≡ Replace (λa b. b = F a) x"

lemma repfun_set : "Set x ⟹ Set (RepFun F x)"
  unfolding RepFun_def by (rule replace_set)

lemma repfunI : "a ∈ x ⟹  F a ∈ RepFun F x"
  unfolding RepFun_def using replace_iff setI by auto

lemma repfun_eqI : "⟦b = F a ; a ∈ x⟧ ⟹ b ∈ RepFun F x"
  using repfunI by auto

lemma repfunE : assumes "Set x" 
  shows "⟦ b ∈ RepFun F x ; 
        ⋀a. ⟦ a ∈ x ; b = F a ⟧ ⟹ P ⟧ ⟹ P"
  by (simp add: RepFun_def replace_iff[OF assms], blast)

lemma repfun_cong : assumes "Set x" "Set y" 
  shows "⟦ x = y ; ⋀a. a ∈ y ⟹ F a = G a ⟧ ⟹ RepFun F x = RepFun G y"
  unfolding RepFun_def using assms by simp

lemma repfun_iff  : assumes "Set x" 
  shows "b ∈ RepFun F x ⟷ (bex x (λc. b = F c))"
  using repfun_eqI repfunE[OF assms] by blast

lemma repfun_union : assumes "Set x"
  shows "b ∈ ⋃ RepFun F x ⟷ (bex x (λa. b ∈ F a))"
  by (simp add: union_iff[OF repfun_set[OF ‹Set x›]] bex_def rex_def
                repfun_iff[OF ‹Set x›], auto)

subsection ‹Intersection of a set›

definition Int :: "'d ⇒ 'd" (‹⋂ _› [90] 90) where
  "⋂ x ≡ Collect (λa. ball x (λy. a ∈ y)) (Union x)"

lemma Int_set : "Set x ⟹ Set (⋂ x)" 
  unfolding Int_def by (rule collect_set[OF union_set])

lemma Int_iff: assumes "Set x" 
  shows "a ∈ ⋂ x ⟷ (ball x (λy. a ∈ y)) ∧ x ≠ ∅ "
proof
  assume "a ∈ ⋂ x" hence "a ∈ ⋃ x" "ball x (λy. a ∈ y)" unfolding Int_def
    using collectE[OF union_set[OF assms]] by auto
  thus "ball x (λy. a ∈ y) ∧ x ≠ ∅" using union_emp[OF assms] by auto
next
  assume "ball x (λy. a ∈ y) ∧ x ≠ ∅" hence ball:"ball x (λy. a ∈ y)" and "x ≠ ∅" by simp_all
  from ‹x ≠ ∅› have "a ∈ ⋃ x"
  proof (rule not_emptyE[OF assms])
    fix y assume "y ∈ x" thus "a ∈ ⋃ x" 
      using ball unionI[OF assms] unfolding ball_def rall_def by auto
  qed
  thus "a ∈ ⋂ x" unfolding Int_def using ball collectI by auto
qed

lemma IntI : assumes "Set x" shows "⟦ ⋀y. y ∈ x ⟹ a ∈ y ; x ≠ ∅ ⟧ ⟹ a ∈ ⋂ x"
  using Int_iff[OF assms] by auto

lemma IntD : "⟦ a ∈ ⋂ x ; y ∈ x ⟧ ⟹ a ∈ y"
  using setI Int_iff by auto

lemma IntE : assumes "Set x" shows "⟦ a ∈ ⋂ x ; y ∉ x ⟹ R; a ∈ y ⟹ R⟧ ⟹ R"
  using Int_iff[OF assms] by auto

subsection ‹Unordered pairs; defined by replacement over 𝒫 (𝒫 ∅)›

abbreviation "φ x y a b ≡ (a = ∅ ∧ b = x) ∨ (a = 𝒫 ∅ ∧ b = y)"

definition upair :: "'d ⇒ 'd ⇒ 'd" where
  "upair x y ≡ Replace (φ x y) (𝒫 (𝒫 Emp))"

lemma upair_set : "Set (upair a b)"
  unfolding upair_def
  by (rule replace_set[OF pow_set[OF pow_set[OF emp_set]]])

lemma pow_emp_iff : "x ∈ 𝒫 ∅ ⟷ x = ∅" unfolding pow_iff[OF emp_set]
proof
  assume "x ⊆ ∅" hence "Set x" "∀a. a ∉ x" using subset_iff by auto
  thus "x = ∅" using equals0I by blast
next
  assume "x = ∅" thus "x ⊆ ∅" using empty_subsetI[OF emp_set] by simp
qed

lemma pow_pow_emp_iff : "x ∈ 𝒫 (𝒫 ∅) ⟷ (x = ∅ | x = 𝒫 ∅)" unfolding pow_iff[OF pow_set[OF emp_set]]
proof
  assume "x ⊆ 𝒫 ∅" hence "Set x" using subset_set by auto (*and "∀a. a ∈ x ⟶ a ∈ 𝒫 ∅" by auto*)
  hence "x = ∅ | (∃y. y ∈ x)" by (rule setE)
  thus "x = ∅ | x = 𝒫 ∅"
  proof (rule, simp)
    assume "∃y. y ∈ x" hence emp_x : "∅ ∈ x" using ‹x ⊆ 𝒫 ∅› pow_emp_iff by auto
    have "x = 𝒫 ∅" proof (rule equality_iffI[OF ‹Set x› pow_set[OF emp_set]])
      fix a show "a ∈ x ⟷ a ∈ 𝒫 ∅" using ‹x ⊆ 𝒫 ∅› ‹∅ ∈ x› pow_emp_iff by auto
    qed
    thus "x = ∅ | x = 𝒫 ∅" by simp
  qed
next
  assume "x = ∅ | x = 𝒫 ∅" thus "x ⊆ 𝒫 ∅" 
  proof
    assume "x = ∅" thus "x ⊆ 𝒫 ∅"
      using empty_subsetI[OF pow_set[OF emp_set]] by simp
  next
    assume "x = 𝒫 ∅" thus "x ⊆ 𝒫 ∅" 
      using subset_refl[OF pow_set[OF emp_set]] by simp
  qed
qed

lemma φ_uniq : "φ x y a b ⟹ (∀z. φ x y a z ⟶ z = b)"
  using pow_emp_iff by blast

thm replace_iff[OF pow_set[OF pow_set[OF emp_set]]]
lemma upair_iff : "b ∈ upair x y ⟷ (b = x | b = y)"
  unfolding upair_def 
proof (rule, erule replaceE2[OF pow_set[OF pow_set[OF emp_set]]])
  fix a assume "a ∈ (𝒫 𝒫 ∅)" and "φ x y a b"
  thus "(b = x | b = y)" using pow_pow_emp_iff by auto
next
  assume "(b = x | b = y)"
  thus "b ∈ Replace (φ x y) (𝒫 𝒫 ∅)" 
  proof
    assume "b = x" show "b ∈ Replace (φ x y) (𝒫 𝒫 ∅)"
    proof (rule replaceI)
      let ?a = "∅" show "φ x y ?a b" using ‹b = x› by simp
      show "∅ ∈ 𝒫 (𝒫 ∅)" using pow_pow_emp_iff by simp
      show "⋀c. φ x y ?a c ⟹ c = b" using φ_uniq[OF `φ x y ?a b`] by auto
    qed
  next
    assume "b = y" show "b ∈ Replace (φ x y) (𝒫 𝒫 ∅)"
    proof (rule replaceI)
      let ?a = "𝒫 ∅" show "φ x y ?a b" using ‹b = y› by simp
      show "𝒫 ∅ ∈ 𝒫 (𝒫 ∅)" using pow_pow_emp_iff by simp
      show "⋀c. φ x y ?a c ⟹ c = b" using φ_uniq[OF `φ x y ?a b`] by auto
    qed
  qed
qed

lemma upairI1 : "a ∈ upair a b" using upair_iff by simp
lemma upairI2 : "b ∈ upair a b" using upair_iff by simp

lemma upairE : "⟦ a ∈ upair b c ; a=b ⟹ P ; a=c ⟹ P⟧ ⟹ P" 
  by (simp add: upair_iff, blast)

subsection ‹Equality for very small sets›

lemma sng_eq_iff : "sng a = sng b ⟷ a=b"
  by (rule equality_iff[OF sng_set, OF sng_set, THEN cnf.iff_trans],
      use sng_iff in auto)

lemma upair_eq_iff : "(upair a b) = (upair c d) ⟷ ((a=c ∧ b=d) | (a=d ∧ b=c))"
proof (rule equality_iff[OF upair_set upair_set, THEN cnf.iff_trans], rule)
  assume eq:"∀x. x ∈ upair a b ⟷ x ∈ upair c d"
  have "a = b | a ≠ b" by simp
  thus "a = c ∧ b = d ∨ a = d ∧ b = c"
  proof
    assume "a = b" hence "upair a b = sng a" unfolding sng_def by simp
    hence "c = a ∧ d = a" using eq upair_iff sng_iff by auto
    thus "a = c ∧ b = d ∨ a = d ∧ b = c" using ‹a = b› by auto
  next
    assume "a ≠ b"
    have "a ∈ upair c d" "b ∈ upair c d" using eq upairI1 upairI2 by auto
    thus "a = c ∧ b = d ∨ a = d ∧ b = c" using ‹a≠b› upair_iff by auto
  qed
next
  assume "a = c ∧ b = d ∨ a = d ∧ b = c" 
  thus "∀x. (x ∈ upair a b) ⟷ (x ∈ upair c d)" using upair_iff by auto
qed

subsection ‹Cartesian product›

definition cprod :: "['d, 'd] ⇒ 'd" (infixr ‹×› 80) where
  "x × y ≡ ⋃ Replace (λa z. z = Replace (λb p. p = kpair a b) y) x"

lemma cprod_set : assumes "Set x"
  shows "Set (x × y)" unfolding cprod_def 
  by (rule union_set[OF replace_set[OF assms]])

lemma cprod_iff : assumes "Set x" "Set y"
  shows "p ∈ x × y ⟷ (∃a b. p = kpair a b ∧ a ∈ x ∧ b ∈ y)" 
proof -  
  let ?R = "Replace (λa z. z = Replace (λb p. p = kpair a b) y) x"
  show ?thesis unfolding cprod_def
  proof (rule union_iff[OF replace_set[OF ‹Set x›], THEN cnf.iff_trans], rule)
    assume "bex ?R (λa. p ∈ a)" then obtain r where "r ∈ ?R" "p ∈ r" by auto
    show "∃a b. p = kpair a b ∧ a ∈ x ∧ b ∈ y"
    proof (rule replaceE2[OF ‹Set x› ‹r ∈ ?R›], rule)
      fix a assume "a ∈ x" "r = Replace (λb p. p = kpair a b) y"
      hence "p ∈ Replace (λb p. p = kpair a b) y" using ‹p ∈ r› by simp
      show "∃b. p = kpair a b ∧ a ∈ x ∧ b ∈ y"
      proof (rule replaceE2[OF ‹Set y› ‹p ∈ Replace (λb p. p = kpair a b) y›], rule)
        fix b assume "b ∈ y" "p = kpair a b" 
        thus "p = kpair a b ∧ a ∈ x ∧ b ∈ y" using ‹a ∈ x› by auto
      qed
    qed
  next
    assume "∃a b. p = kpair a b ∧ a ∈ x ∧ b ∈ y"
    then obtain a b where "p = kpair a b" "a ∈ x" "b ∈ y" by auto
    hence r: "p ∈ Replace (λb p. p = kpair a b) y" 
             "Replace (λb p. p = kpair a b) y ∈ ?R" 
      by (auto intro: replaceI)
    thus "bex ?R (λr. p ∈ r)" using r by auto
  qed
qed

lemma cprod_iff_pair : assumes "Set x" "Set y" 
  shows "kpair a b ∈ x × y ⟷ a ∈ x ∧ b ∈ y" 
  using cprod_iff[OF assms] pair_inject by blast

lemma cprodI : "a ∈ x ⟹ b ∈ y ⟹ p = kpair a b ⟹ p ∈ x × y"
  using setI cprod_iff by auto

lemma cprodI_pair : "a ∈ x ⟹ b ∈ y ⟹ kpair a b ∈ x × y"

lemma cprodD1 : assumes "Set x" "Set y" shows "kpair a b ∈ x × y ⟹ a ∈ x"

lemma cprodD2 : assumes "Set x" "Set y" shows "kpair a b ∈ x × y ⟹ b ∈ y"

lemma cprodE : assumes "Set x" "Set y" 
  shows "⟦ p ∈ x × y ; ⋀a b. ⟦ a ∈ x ; b ∈ y ; p = kpair a b ⟧ ⟹ P ⟧ ⟹ P"
